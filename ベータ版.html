    function updateUI() {

        document.getElementById('display-total').innerText = state.total.toLocaleString();

        let remaining = state.total;

        const cats = [

            {id:'park', g:state.goals.park}, {id:'hotel', g:state.goals.hotel},

            {id:'food', g:state.goals.food}, {id:'dpa', g:state.goals.dpa},

        ];


        let totalGoalSum = 0;

        cats.forEach(c => {

            const current = Math.min(remaining, c.g);

            const p = c.g > 0 ? (current / c.g) * 100 : 0;

    // --- 花火・演出エンジン ---

    const canvas = document.getElementById('fireworks-canvas');

    const ctx = canvas.getContext('2d');

    let particles = [];


    function startFireworks() {

        canvas.style.display = 'block';

        canvas.style.opacity = '1'; // 表示時に透明度を1にする

        const duration = 5000;

        const endTime = Date.now() + duration;

        

        function loop() {

            if (Date.now() < endTime) {

                if (Math.random() < 0.2) createBurst();

                if (Math.random() < 0.15) createSidePyro();

                draw();

                requestAnimationFrame(loop);

            } else {

                // スッと消えるアニメーション開始

                canvas.style.opacity = '0';

                setTimeout(() => {

                    canvas.style.display = 'none';

                    particles = [];

                }, 1000); // CSSのtransition(1s)と合わせる

            }

        }

        loop();

    }


    function createSidePyro() {

        const side = Math.random() < 0.5 ? 'left' : 'right';

        const x = side === 'left' ? 0 : canvas.width;

        const y = canvas.height;

        const vx = side === 'left' ? (Math.random() * 8 + 5) : -(Math.random() * 8 + 5);

        const vy = -(Math.random() * 12 + 10);

        const hue = Math.random() < 0.5 ? 45 : 200;


        for(let i=0; i<3; i++) {

            particles.push({

                x, y, vx: vx + (Math.random()-0.5), vy: vy + (Math.random()-0.5),

                r: 4, color: `hsl(${hue}, 100%, 70%)`, alpha: 1, decay: 0.015, isPyro: true

            });

        }

    }


    function createBurst() {

        const x = Math.random() * canvas.width;

        const y = Math.random() * canvas.height * 0.7;

        const hue = Math.random() * 360;

        for (let i = 0; i < 80; i++) {

            particles.push({

                x, y, vx: (Math.random() - 0.5) * 14, vy: (Math.random() - 0.5) * 14,

                r: Math.random() * 2 + 1, color: `hsl(${hue + (Math.random()*40 - 20)}, 100%, 60%)`,

                alpha: 1, decay: Math.random() * 0.02 + 0.01, isPyro: false

            });

        }

    }


    function draw() {

        ctx.fillStyle = 'rgba(0, 11, 26, 0.25)';

        ctx.fillRect(0, 0, canvas.width, canvas.height);

        particles.forEach((p, i) => {

            p.x += p.vx; p.y += p.vy;

            p.vy += p.isPyro ? 0.25 : 0.15;

            p.alpha -= p.decay;

            ctx.globalAlpha = p.alpha;

            ctx.shadowBlur = p.isPyro ? 15 : 0;

            ctx.shadowColor = p.color;

            ctx.fillStyle = p.color;

            ctx.beginPath(); ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2); ctx.fill();

            if (p.alpha <= 0) particles.splice(i, 1);

        });

        ctx.shadowBlur = 0;

    }

        canvas.height = window.innerHeight;

    });

    canvas.height = window.innerHeight;
